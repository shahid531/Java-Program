import java.util.*;

public class Hello {
    public static void main(String[] args){ 
        int[] arr1 = {2,5,3,8,4,1,9};
        for(int i = arr1.length-1; i > 0; i--){
            for(int j = 0; j < i; j++){
                if(arr1[j] > arr1[j+1]){
                    int temp = arr1[j];
                    arr1[j] = arr1[j+1];
                    arr1[j+1] = temp;
                }
            }
        }
        System.out.println(Arrays.toString(arr1));
    }
}
